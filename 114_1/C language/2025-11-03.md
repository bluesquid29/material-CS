```c
#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    int i, score[4];

    score[0] = 78;
    score[1] = 55;
    /* score[2] = 92; */
    score[3] = 80;
    

    for ( i = 0; i <= 4; i++)
    {
        printf("score[%d] = %d\n", i, score[i] );
    }
    
    return 0;
}
```

## sizeof

review:

| Data type                       |                      | bytes | range       |               |
| ------------------------------- | -------------------- | ----- | ----------- | ------------- |
| character                       | `char`               | 1     | -128        | 127           |
|                                 | `unsigned char`      |       | 0           | 255           |
|                                 |                      |       |             |               |
| short integer                   | `short`              | 2     | -32768      | 32767         |
|                                 | `unsigned short`     |       | 0           | 65535         |
| integer                         | `int`                | 4     | -2147483648 | 2,147,483,647 |
|                                 | `unsigned int`       |       | 0           | 4,294,967,295 |
| long integer                    | `long`               | 4     | (`int`)     |               |
|                                 | `unsigned long`      |       |             |               |
|                                 | `long long`          | 8     | (...)       |               |
|                                 | `unsigned long long` |       |             |               |
|                                 |                      |       |             |               |
| single-precision floating point | `float`              | 4     | 1.2e-038    | 3.4e+038      |
| double-precision floating point | `double`             | 8     | 2.2e-308    | 1.8e+308      |
|                                 | `long double`        | 16    | (...)       |               |

```c
#include <stdio.h>

int main() {
    char arrC[5] = {'a', 'b', 'c', 'd', 'e'};
    short arrD[5] = {11, 12, 13, 14, 15};
    int arrA[5] = {1, 2, 3, 4, 5};
    long long arrE[5] = {16, 17, 18, 19, 20};
    float arrF[5] = {21.1, 22.2, 23.3, 24.4, 25.5};
    double arrB[5] = {6, 7, 8, 9, 10};
    
    // char
    printf("Size of char array: %lu bytes\n", sizeof(arrC));
    printf("Size of element in char array: %lu bytes\n", sizeof(arrC[0]));
    printf("Number of elements in char array: %lu\n\n", sizeof(arrC) / sizeof(arrC[0]));

    // short
    printf("Size of short array: %lu bytes\n", sizeof(arrD));
    printf("Size of element in short array: %lu bytes\n", sizeof(arrD[0]));
    printf("Number of elements in short array: %lu\n\n", sizeof(arrD) / sizeof(arrD[0]));

    // int
    printf("Size of int array: %lu bytes\n", sizeof(arrA));
    printf("Size of element in int array: %lu bytes\n", sizeof(arrA[0]));
    printf("Number of elements in int array: %lu\n\n", sizeof(arrA) / sizeof(arrA[0]));

    // long long
    printf("Size of long long array: %lu bytes\n", sizeof(arrE));
    printf("Size of element in long long array: %lu bytes\n", sizeof(arrE[0]));
    printf("Number of elements in long long array: %lu\n\n", sizeof(arrE) / sizeof(arrE[0]));

    // float
    printf("Size of float array: %lu bytes\n", sizeof(arrF));
    printf("Size of element in float array: %lu bytes\n", sizeof(arrF[0]));
    printf("Number of elements in float array: %lu\n\n", sizeof(arrF) / sizeof(arrF[0]));

    // double
    printf("Size of double array: %lu bytes\n", sizeof(arrB));
    printf("Size of element in double array: %lu bytes\n", sizeof(arrB[0]));
    printf("Number of elements in double array: %lu\n\n", sizeof(arrB) / sizeof(arrB[0]));

    return 0;
}

```

```
Size of char array: 5 bytes
Size of element in char array: 1 bytes
Number of elements in char array: 5

Size of short array: 10 bytes
Size of element in short array: 2 bytes
Number of elements in short array: 5

Size of int array: 20 bytes
Size of element in int array: 4 bytes
Number of elements in int array: 5

Size of long long array: 40 bytes
Size of element in long long array: 8 bytes
Number of elements in long long array: 5

Size of float array: 20 bytes
Size of element in float array: 4 bytes
Number of elements in float array: 5

Size of double array: 40 bytes
Size of element in double array: 8 bytes
Number of elements in double array: 5
```

Because `sizeof` returns a value of type `size_t`, which is an unsigned long. Using `%lu` ensures correct formatting and avoids warnings on some compilers.
